// 1 – Вивід заголовка console.log(document.getElementById("header").innerText); // 2 – Зміна тексту document.getElementById("changeTextButton").onclick = function() { document.getElementById("text").innerText = "Текст змінено!"; }; // 3 – Додавання абзацу document.getElementById("addParagraphButton").onclick = function() { const p = document.createElement("p"); p.innerText = "Новий абзац!"; document.body.appendChild(p); }; // 4 – Видалення останнього абзацу document.getElementById("removeLastParagraph").onclick = function() { const paragraphs = document.getElementsByTagName("p"); if (paragraphs.length > 0) { paragraphs[paragraphs.length - 1].remove(); } }; // 5 – Зміна фону document.getElementById("changeBackgroundColor").onclick = function() { document.body.style.backgroundColor = "#d9e6ff"; }; // 6 – Hover ефект const hoverText = document.getElementById("hoverText"); hoverText.onmouseover = () => hoverText.style.color = "red"; hoverText.onmouseout = () => hoverText.style.color = "black"; // 7 – Перевірка email document.getElementById("emailForm").onsubmit = function(event) { const email = document.getElementById("email").value; const pattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!pattern.test(email)) { alert("Email невірний!"); event.preventDefault(); } else { alert("Email коректний!"); } }; // 8 – Зміна зображення document.getElementById("changeImage").onclick = function() { const img = document.getElementById("image"); if (img.src.includes("Image1")) { img.src = "https://via.placeholder.com/150x80?text=Image2"; } else { img.src = "https://via.placeholder.com/150x80?text=Image1"; } }; // 9 – Додавання класу document.getElementById("addClassButton").onclick = function() { document.getElementById("styledText").classList.add("red"); };
